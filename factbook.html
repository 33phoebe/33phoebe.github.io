q6 = '''
SELECT name, population/area_land pop_density FROM facts 
WHERE  (name != 'World' AND name !='Antarctica')
ORDER BY pop_density DESC
LIMIT 30'''

df = pd.read_sql_query(q6, conn)
df = df.set_index('name')
print(df.head())
